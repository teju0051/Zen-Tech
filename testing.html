int main() {
    double a, b, c, discriminant, root1, root2, realPart, imaginaryPart;

    // Input the coefficients a, b, and c
    printf("Enter coefficients a, b, and c: ");
    scanf("%lf %lf %lf", &a, &b, &c);

    // Calculate the discriminant
    discriminant = b * b - 4 * a * c;

    // Check the nature of the discriminant and calculate the roots
    if (discriminant > 0) {
        // Two real and distinct roots
        root1 = (-b + sqrt(discriminant)) / (2 * a);
        root2 = (-b - sqrt(discriminant)) / (2 * a);
        printf("Roots are real and distinct: %.2lf and %.2lf\n", root1, root2);
    } 
    else if (discriminant == 0) {
        // Two real and identical roots
        root1 = -b / (2 * a);
        printf("Roots are real and identical: %.2lf\n", root1);
    } 
    else {
        // Complex (imaginary) roots
        realPart = -b / (2 * a);
        imaginaryPart = sqrt(-discriminant) / (2 * a);
        printf("Roots are complex: %.2lf + %.2lfi and %.2lf - %.2lfi\n", 
               realPart, imaginaryPart, realPart, imaginaryPart);
    }

    return 0;
}

void fibonacci(int n, int firstTerm, int secondTerm) {
    if (n > 0) {
        printf("%d, ", firstTerm);
        fibonacci(n - 1, secondTerm, firstTerm + secondTerm); // Recursive call
    }

unsigned long long factorial(int n) {
    if (n == 0 || n == 1) // Base case: 0! and 1! both equal 1
        return 1;
    else

}

#include <stdio.h>

void bubbleSort(int arr[], int n, int ascending) {
  for (int i = 0; i < n - 1; i++) {
      for (int j = 0; j < n - i - 1; j++) {
          // Swap based on the order selected
          if ((ascending && arr[j] > arr[j + 1]) || (!ascending && arr[j] < arr[j + 1])) {
              // Swap arr[j] and arr[j+1]
              int temp = arr[j];
              arr[j] = arr[j + 1];
              arr[j + 1] = temp;
          }
      }
  }
}

void printArray(int arr[], int n) {
  for (int i = 0; i < n; i++) {
      printf("%d ", arr[i]);
  }
  printf("\n");
}

int main() {
  int n;

  // Input number of elements in the array
  printf("Enter the number of elements: ");
  scanf("%d", &n);

  int arr[n]; // Declare array of size n

  // Input elements into the array
  printf("Enter %d elements:\n", n);
  for (int i = 0; i < n; i++) {
      scanf("%d", &arr[i]);
  }

  int choice;
  printf("Sort in (1) Ascending or (2) Descending order? Enter 1 or 2: ");
  scanf("%d", &choice);

  // Perform sorting based on user choice
  if (choice == 1) {
      bubbleSort(arr, n, 1); // Ascending
      printf("Sorted array in ascending order:\n");
  } else if (choice == 2) {
      bubbleSort(arr, n, 0); // Descending
      printf("Sorted array in descending order:\n");
  } else {
      printf("Invalid choice!\n");
      return 1; // Exit with an error
  }

  // Print the sorted array
  printArray(arr, n);
  return 0;
}

  // Function to check if a string is a palindrome
  int isPalindrome(char str[]) {
      int left = 0;
      int right = strlen(str) - 1;
  
      while (left < right) {
          // Ignore non-alphanumeric characters and case
          while (left < right && !isalnum(str[left])) left++;
          while (left < right && !isalnum(str[right])) right--;
  
          // Compare characters
          if (tolower(str[left]) != tolower(str[right])) {
              return 0; // Not a palindrome
          }
          left++;
          right--;
      }
      return 1; // Is a palindrome
  }
  
  int main() {
      char str[MAX_LENGTH];
  
      // Input the string
      printf("Enter a string: ");
      fgets(str, sizeof(str), stdin);

      // Check if the string is a palindrome

          printf("'%s' is a palindrome.\n", str);

#include <stdio.h>

// Function to swap two numbers using call-by-value
void swapByValue(int a, int b) {
    int temp;
    temp = a; 
    a = b; 
    b = temp; 
    printf("Inside swapByValue: a = %d, b = %d\n", a, b);
}

// Function to swap two numbers using call-by-reference
void swapByReference(int *a, int *b) {
    int temp;
    temp = *a; 
    *a = *b; 
    *b = temp; 
    printf("Inside swapByReference: *a = %d, *b = %d\n", *a, *b);
}

int main() {
    int x, y;

    // Input two numbers
    printf("Enter two numbers (x and y): ");
    scanf("%d %d", &x, &y);

    // Call swapByValue
    printf("Before swapByValue: x = %d, y = %d\n", x, y);
    swapByValue(x, y); // Call-by-value
    printf("After swapByValue: x = %d, y = %d (no change)\n\n", x, y);

    // Call swapByReference
    printf("Before swapByReference: x = %d, y = %d\n", x, y);
    swapByReference(&x, &y); // Call-by-reference
    printf("After swapByReference: x = %d, y = %d (values swapped)\n", x, y);

    return 0;
}

// Function to multiply two matrices
void multiplyMatrices(int first[MAX][MAX], int second[MAX][MAX], int result[MAX][MAX], int r1, int c1, int r2, int c2) {
    // Initialize result matrix to 0
    for (int i = 0; i < r1; i++) {
        for (int j = 0; j < c2; j++) {
            result[i][j] = 0;
        }
    }

    // Perform matrix multiplication
    for (int i = 0; i < r1; i++) {
        for (int j = 0; j < c2; j++) {
            for (int k = 0; k < c1; k++) {
                result[i][j] += first[i][k] * second[k][j];
            }
        }
    }
}

int main() {

            scanf("%d", &first[i][j]);
        }

              printf("Amount deposited successfully! New balance: %.2f\n", accounts[accountNumber - 1].balance);
          } else {
              printf("Invalid amount!\n");
          }
      } else {
          printf("Invalid account number!\n");
      }
  }
  
  // Function to withdraw money
  void withdrawMoney(struct Account accounts[], int count) {
      int accountNumber;
      float amount;
      printf("Enter your account number: ");
      scanf("%d", &accountNumber);
  
      if (accountNumber > 0 && accountNumber <= count) {
          printf("Enter amount to withdraw: ");
          scanf("%f", &amount);
          if (amount > 0 && amount <= accounts[accountNumber - 1].balance) {
              accounts[accountNumber - 1].balance -= amount;
              printf("Amount withdrawn successfully! New balance: %.2f\n", accounts[accountNumber - 1].balance);
          } else {
              printf("Invalid amount or insufficient balance!\n");
          }
      } else {
          printf("Invalid account number!\n");
      }
  }
